9) check if the number is an Armstrong number or not

	n = input("enter the number :")
	power = len(n)
	output = 0
	for i in n:
    		a = int(i)**power
    		output+=a
	if output == int(n):
    		print("it a Armstrong number")
	else:
    		print("its not a Armstrong number")

	output : 
	
	enter the number :153
	it a Armstrong number
